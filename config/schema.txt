CREATE TABLE IF NOT EXISTS nonces (
  nonce TEXT,
  ip TEXT UNIQUE,
  endpoint TEXT,
  time TIMESTAMP
);

CREATE TABLE IF NOT EXISTS users (
  userid TEXT UNIQUE,
  username TEXT,
  email TEXT UNIQUE,
  pass TEXT
);

CREATE TABLE IF NOT EXISTS help_pages (
  id TEXT UNIQUE,
  file TEXT UNIQUE,
  edited TIMESTAMP
);

CREATE TABLE IF NOT EXISTS blog_posts (
  id TEXT UNIQUE,
  file TEXT UNIQUE,
  summary TEXT,
  title TEXT,
  image TEXT,
  pfp TEXT,
  author TEXT,
  edited TIMESTAMP
);

DROP TABLE sessions;
CREATE TABLE IF NOT EXISTS sessions (
  token TEXT UNIQUE,
  username TEXT,
  userid TEXT
)
