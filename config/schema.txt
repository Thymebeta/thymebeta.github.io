CREATE TABLE IF NOT EXISTS nonces (
  nonce TEXT,
  ip TEXT UNIQUE,
  endpoint TEXT,
  time TIMESTAMP
);

CREATE TABLE IF NOT EXISTS users (
  userid TEXT UNIQUE,
  username TEXT,
  email TEXT UNIQUE,
  pass TEXT
);

CREATE TABLE IF NOT EXISTS help_pages (
  id TEXT UNIQUE,
  file TEXT UNIQUE,
  edited TIMESTAMP
);

CREATE TABLE IF NOT EXISTS blog_posts (
  id TEXT UNIQUE,
  file TEXT UNIQUE,
  summary TEXT,
  title TEXT,
  image TEXT,
  pfp TEXT,
  author TEXT,
  edited TIMESTAMP
);

CREATE TABLE IF NOT EXISTS sessions (
  token TEXT UNIQUE,
  username TEXT,
  userid TEXT
);

CREATE TABLE IF NOT EXISTS upload_keys (
  key TEXT UNIQUE,
  url TEXT UNIQUE,
  session_token TEXT,
  expires TIMESTAMP
);

CREATE TABLE IF NOT EXISTS videos (
  id TEXT UNIQUE,
  poster TEXT,
  title TEXT,
  description TEXT,
  tags TEXT,
  public INTEGER,
  license TEXT,
  uploaded TIMESTAMP,
  sources TEXT
)
